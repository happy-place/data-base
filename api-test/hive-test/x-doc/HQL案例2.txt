SELECT
	temp.empno,

	temp.sal,
	temp.deptno
FROM(
	SELECT

		sal,
		deptno,
		ROW_NUMBER() OVER(PARTITION BY deptno ORDER BY sal DESC) rank
	FROM
			emp) temp
WHERE
temp.rank <= 2

统计某个网站某天的所有PV数据
格式：

2015-08-28	35000

SELECT
	temp.date,
	COUNT(temp.url) pv
FROM(
	SELECT
	SUBSTRING(trackTime, 0, 10) date,
	url
	FROM
	db_web_data.track_log
WHERE
	LENGTH(url) > 0) temp
GROUP BY
	temp.date;

统计某个网站某天的所有UV数据
格式：
2015-08-28	35000  12000
SELECT
	temp.date,
	COUNT(temp.url) pv,
	COUNT(DISTINCT temp.guid) uv
	FROM(
	SELECT
		SUBSTRING(trackTime, 0, 10) date,
		url,
		guid
	FROM
		db_web_data.track_log
	WHERE
	length(url) > 0) temp
GROUP BY
temp.date;
